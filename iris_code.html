3 Accessing Quality of Data

To access the quality of the Iris Data set, I will perform check on the completeness (check for empty entries) , uniqueness (check for duplicates) and validity (check for correct data type) of the data set.

cleanDataset <- function(iris){
  # Check for completeness (check for empty data entries and exclude row if empty entry exists) 
  iris %>% mutate_all(na_if," ") %>% na.omit() # change all empty entries to "NA" and exclude those rows for analysis
  # Check/remove duplicates (eg. row 102 and 143 are the same)
  iris <- iris %>% distinct()
  # Check for validity (data should be numeric)
  iris <- iris[!is.na(as.numeric(iris$Sepal.Length)&as.numeric(iris$Sepal.Width)& # Delete rows; only for feature data, not label data
                          as.numeric(iris$Petal.Length)&as.numeric(iris$Petal.Width)), ]    #because incorrect label data does not affect how we identify the 10 most similar flowers 
}

9 Using Euclidean distance as a form of similarity measure

In order to retrieve records of the ten most similar data points in the existing Iris data to the input arguments, I have used one of the most common similarity measures, which is through the use of euclidean distance. I will first combine the input argument with the existing Iris data, then i will calculate the euclidean distances between every 2 data points, storing all the results in a n by n matrix, where n = (number of data points in the iris data set + input data) Thereafter i will extract the nth row (which consists of the euclidean distance of the input data from each of the existing data point), and sort it in ascending order. The “nearest10_euclidDist” function will return the indexes of the 10 most similar data points from the iris dataset.

# A function that returns the index of the 10 most similar flowers to the new iris flower measured

nearest10_euclidDist <- function(dataset,input){
  dataset_temp <- subset(dataset,select = -c(Species)) # remove the Species column from iris dataset
  dataset_temp <- rbind(dataset_temp,input) # add the new input into the (iris) dataset 
  numrows <- nrow(dataset_temp) # position of the new input data
  euclidDist_matrix <- as.matrix(dist(dataset_temp[, 1:4],    # returns the euclidian matrix; n by n
                              method = "euclidean", 
                              upper = TRUE, 
                              diag = TRUE)) 
  numrows <- nrow(euclidDist_matrix) # position of the input iris flower
  euclidDist_ptOfInterest <- euclidDist_matrix[numrows,] # i am interested in this flower, which are the 10 most similar flowers?
  #print(euclidDist_ptOfInterest)
  indexes <- order(euclidDist_ptOfInterest, decreasing = FALSE) # order them by ASC order, index (smaller dist, more similar)
  indexes <- indexes[!indexes %in% c(numrows)] # exclude the point of interest (input data)
  first10_indexes <- indexes[1:10] # retrieve the first 10 most similar flower (highest cos similarity)
  first10_indexes
}

17 (ADDITIONAL) END OF ANALYSIS; Using Cosine similairity as a form of similarity measure

I have also also tried using Cosine Similarity as a form of similarity measure but ultimately chosen to go with euclidean distance as the measure to be used for this test.

# Besides using euclidean distance as a form of similarity measure, i have also tried using cosine similarity 

cosineSim <- function(dataset,input){
  dataset <- subset(dataset,select = -c(Species)) # remove the Species column from iris dataset, causing some NA issue
  dataset <- rbind(dataset,input) # add the new input into the (iris) dataset 
  numrows <- nrow(dataset)
  cosineSimMatrix <- matrix(0,numrows,numrows) # create a zero matrix for storing cosine sim value; dim should be n by n; n = num obs in dataset
  #print(dim(cosineSimMatrix))
  for (row in 1:numrows){
    for (col in 1:numrows){
      #print(row)
      #print(col)
      firstVec <- as.numeric(dataset[row,1:4]) # 1 row of dataset
      #print(firstVec)
      secondVec <- as.numeric(dataset[col,1:4]) # another row of dataset with comparison to the firstVec
      #print(secondVec)
      cos_theta <- dot(firstVec,secondVec)/(sqrt(dot(firstVec,firstVec))*sqrt(dot(secondVec,secondVec))) # cos sim measure
      cosineSimMatrix[row,col] <- cos_theta # store in the matrix created earlier
}
  }
  cosineSimMatrix
}

#input <- c(5,3.4,1.5,0.2)
#matrix_cosSim <- cosineSim(cleanedIris,input)

nearest10_cosSim <- function(dataset,input){
  cosineSim_matrix <- cosineSim(dataset,input) # returns the cos sim matrix; n by n
  numrows <- nrow(cosineSim_matrix) # position of the input iris flower
  cosineSim_ptOfInterest <- cosineSim_matrix[numrows,] # i am interested in this flower, which are the 10 most similar flowers?
  #print(cosineSim_ptOfInterest)
  indexes <- order(cosineSim_ptOfInterest, decreasing = TRUE) # order them by descending order, index
  #print(indexes)
  indexes <- indexes[!indexes %in% c(numrows)] # exclude the point of interest (input data)
  #print(indexes)
  first10_indexes <- indexes[1:10] # retrieve the first 10 most similar flower (highest cos similarity)
  first10_indexes
}
